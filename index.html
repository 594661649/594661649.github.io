document.getElementById('f').onchange = async (e) => { const file = e.target.files[0]; const blob = await pdf2docx.convert(file); // 生成地址 const url = URL.createObjectURL(blob); const btn = document.getElementById('down'); btn.style.display = 'inline-block'; btn.onclick = () => { const a = document.createElement('a'); a.href = url; a.download = file.name.replace('.pdf','.docx'); a.click(); }; };
